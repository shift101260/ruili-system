// 修正後的檔案上傳函式 async function uploadFileToSupabaseStorage(file, caseId) { try { // 1. 強制確認 caseId 存在，否則歸類到 uncategorized const folderName = (caseId && caseId.trim() !== "") ? caseId.trim() : 'uncategorized'; // 2. 清理檔名中的特殊字元與空格，避免上傳失敗 const safeFileName = file.name.replace(/[^a-zA-Z0-9.\-_]/g, '_'); // 3. 建立明確的動態路徑：案件ID/時間戳記_檔名 const filePath = `${folderName}/${Date.now()}_${safeFileName}`; console.log("正在上傳至路徑：", filePath); // 可以在瀏覽器 Console 控制台檢查上傳路徑 const { data, error } = await supabaseClient.storage.from('ruili-files').upload(filePath, file, { cacheControl: '3600', upsert: false }); if (error) { console.error("Supabase Storage 上傳失敗:", error.message); // 上傳失敗時的備用方案 return { url: URL.createObjectURL(file), path: filePath }; } // 取得公開下載/預覽網址 const { data: publicData } = supabaseClient.storage.from('ruili-files').getPublicUrl(filePath); return { url: publicData.publicUrl, path: filePath }; } catch (err) { console.error("Upload error:", err); return { url: URL.createObjectURL(file), path: "" }; } }
